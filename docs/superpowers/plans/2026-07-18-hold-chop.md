- config.gather 增 `moveSpeedFactor: 0.6`
- types: `PlayerAction = 'idle' | 'walking'`；PlayerState 增 `gathering: boolean`；IntentInput 增 `aimFacing: 0 | 1 | -1`；initialSim 补 `gathering: false`
- player.ts 重写（要点）:
```ts
const moving = input.moveX !== 0 || input.moveY !== 0
const before = p.action
let { facing, action, actionT, gathering, gatherT, pendingFacingT } = p
// 移动基态
const next = moving ? 'walking' : 'idle'
if (next !== action) { action = next; actionT = 0 }
actionT += dt
// 采集通道（正交）；起手与衔接边界取 aimFacing
if (!gathering) {
  if (input.interact) {
    gathering = true; gatherT = 0
    if (input.aimFacing) { facing = input.aimFacing; pendingFacingT = 0 }
  }
} else {
  gatherT += dt
  if (gatherT >= CONFIG.gather.duration) {
    if (input.interact) {
      gatherT -= CONFIG.gather.duration
      if (input.aimFacing) { facing = input.aimFacing; pendingFacingT = 0 }
    } else { gathering = false; gatherT = 0 }
  }
}
// 位移（采集减速）
if (moving) { const k = gathering ? CONFIG.gather.moveSpeedFactor : 1; /* 归一化*speed*k*dt+夹紧 */ }
// 朝向防抖仅在非采集时生效
if (!gathering) { /* 原防抖逻辑 */ } else { pendingFacingT = 0 }
```
